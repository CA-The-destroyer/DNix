Resources
| where type =~ "microsoft.compute/virtualmachines"
| where name has "yourpattern"   // replace with your hostname pattern
| extend nics = properties.networkProfile.networkInterfaces[0].id
| join kind=leftouter (
    Resources
    | where type =~ "microsoft.network/networkinterfaces"
    | mv-expand ipconfig = properties.ipConfigurations
    | extend privateIp = tostring(ipconfig.properties.privateIPAddress)
    | project nicId = id, privateIp
) on $left.nics == $right.nicId
| project HostWithColon = strcat(name, ":"), Blank="", PrivateIPWithColon = strcat(privateIp, ":")
