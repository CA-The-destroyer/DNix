Resources
| where type == "microsoft.compute/virtualmachines"
| where name contains "PATTERN"   // replace PATTERN with your match string
| project vmId = id, vmName = name
| join kind=leftouter (
    Resources
    | where type == "microsoft.network/networkinterfaces"
    | mv-expand ipconfig = properties.ipConfigurations
    | project vmId = tostring(properties.virtualMachine.id), 
              privateIp = tostring(ipconfig.properties.privateIPAddress)
) on vmId
| project Output = strcat(vmName, ":", " ", privateIp, ":")
